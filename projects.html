            using System.Collections;
            using System.Collections.Generic;
            using UnityEngine;
            
            public class SliceSpawner : MonoBehaviour
            {
                public GameObject softSlice;
                public GameObject hardSlice;
                private GameObject[] slices = new GameObject[8];
                private int emptyIndex1;
                private int emptyIndex2;
                private int softSliceIndex;
                private int hardSliceIndex1;
                private int hardSliceIndex2;
                public GameObject parentGameObject;
                
                // Update is called once per frame
                void Update()
                {
                    int spawnAngle = 45;
            
                    while (emptyIndex1 == emptyIndex2 || emptyIndex1 == softSliceIndex || emptyIndex1 == hardSliceIndex1 || emptyIndex1 == hardSliceIndex2 || emptyIndex2 == softSliceIndex || emptyIndex2 == hardSliceIndex1 || emptyIndex2 == hardSliceIndex2 || hardSliceIndex1 == hardSliceIndex2)
                    {
                        emptyIndex1 = Random.Range(0, 8);
                        emptyIndex2 = Random.Range(0, 8);
                        softSliceIndex = Random.Range(0, 8);
                        hardSliceIndex1 = Random.Range(0, 8);
                        hardSliceIndex2 = Random.Range(0, 8);
                    }
                    
            
                    for (int i = 0; i < 8; i++)
                    {
                        if (i == emptyIndex1)
                        {
                            continue;
                        }
                        else if (i == emptyIndex2)
                        {
                            continue;
                        }
            
                        if (i == softSliceIndex)
                        {
                            slices[i] = Instantiate(softSlice, transform.position, Quaternion.Euler(90, 0, i * spawnAngle));
                            transform.SetParent(parentGameObject.transform);
                        }
                        else if (i == hardSliceIndex1)
                        {
                            slices[i] = Instantiate(hardSlice, transform.position, Quaternion.Euler(90, 0, i * spawnAngle));
                            transform.SetParent(parentGameObject.transform);
                        }
                        else if (i == hardSliceIndex2)
                        {
                            slices[i] = Instantiate(hardSlice, transform.position, Quaternion.Euler(90, 0, i * spawnAngle));
                            transform.SetParent(parentGameObject.transform);
                        }
                        else
                        {
                            slices[i] = Instantiate(softSlice, transform.position, Quaternion.Euler(90, 0, i * spawnAngle));
                            transform.SetParent(parentGameObject.transform);
                        }
                    }
                }
            }
